# EXTRACTING CHEBI TERMS FROM PRE-ANALYZED URIs 
## CHEBI
sh run.sh robot extract --method MIREOT \
    --input mirror/chebi.owl \
    --lower-terms imports/chebi_terms.txt \
    --output imports/chebi_import.owl

## IAO
sh run.sh robot extract --method MIREOT \
    --input mirror/iao.owl \
    --lower-terms imports/iao_terms.txt \
    --output imports/iao_import.owl

## STATO
sh run.sh robot extract --method MIREOT \
    --input mirror/stato.owl \
    --lower-terms imports/stato_terms.txt \
    --output imports/stato_import.owl


## STATO - NO INTERMEDIATE
sh run.sh robot extract --method MIREOT \
    --input mirror/stato.owl \
    --lower-terms imports/stato_terms.txt \
    --intermediates none \
    --output imports/stato_import_no_intermediate.owl
